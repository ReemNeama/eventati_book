static List<Map<String, dynamic>> getFeaturesToCompare(String serviceType) {
  switch (serviceType) {
    case 'Venue':
      return _getVenueFeatures();
    case 'Catering':
      return _getCateringFeatures();
    case 'Photographer':
      return _getPhotographerFeatures();
    case 'Planner':
      return _getPlannerFeatures();
    default:
      return [];
  }
}
